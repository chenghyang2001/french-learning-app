# Task: 建立閃卡 Session 前端邏輯（flashcard.js）

## Description

建立 `public/js/flashcard.js`，實作完整的閃卡複習 Session 流程：

1. 頁面載入時呼叫 `GET /api/words?due=true` 取得今日待複習單字
2. 依序顯示每張卡片（正面法文）
3. 使用者點「翻面」後顯示中文與「認識 / 不認識」按鈕
4. 按鈕後呼叫 `PATCH /api/words/:id/review`，記錄結果並跳下一張
5. 所有卡片完成後顯示摘要（認識 N / 不認識 M）

## Acceptance Criteria

- [ ] 頁面載入後自動 fetch 今日待複習單字
- [ ] 無待複習單字時顯示「今天的複習已完成！🎉」（不顯示卡片）
- [ ] 進度顯示「第 X / N 張」
- [ ] 翻面後「認識 / 不認識」按鈕出現，翻面按鈕消失
- [ ] 按「認識」→ PATCH `{ known: true }`，按「不認識」→ PATCH `{ known: false }`
- [ ] API 失敗時在頁面顯示錯誤訊息（不只 console.log）
- [ ] 完成後顯示 `#summary`，隱藏卡片與按鈕

## Technical Details

```javascript
// public/js/flashcard.js（ESM module，flashcard.html 已用 type="module" 載入）

const $ = (id) => document.getElementById(id);

let queue = [];       // 待複習單字陣列
let current = 0;      // 目前卡片索引
let knownCount = 0;
let unknownCount = 0;

async function init() {
  try {
    const res = await fetch("/api/words?due=true");
    const data = await res.json();
    queue = data.words ?? [];

    if (queue.length === 0) {
      $("progress").textContent = "今天的複習已完成！🎉";
      $("card-container").classList.add("hidden");
      $("controls").classList.add("hidden");
      return;
    }
    showCard(0);
  } catch (err) {
    $("progress").textContent = `載入失敗：${err.message}`;
  }
}

function showCard(index) {
  const word = queue[index];
  $("progress").textContent = `第 ${index + 1} / ${queue.length} 張`;
  $("word-french").textContent = word.french;
  $("word-pos").textContent = word.cefr_level ?? "";
  $("word-english").textContent = word.english;

  // 重置翻轉狀態
  $("card-inner").classList.remove("flipped");
  $("btn-flip").style.display = "";
  $("review-buttons").style.display = "none";
}

$("btn-flip").addEventListener("click", () => {
  $("card-inner").classList.add("flipped");
  $("btn-flip").style.display = "none";
  $("review-buttons").style.display = "flex";
});

async function handleReview(known) {
  const word = queue[current];
  try {
    await fetch(`/api/words/${word.id}/review`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ known }),
    });
  } catch (err) {
    $("progress").textContent = `記錄失敗：${err.message}`;
    return;
  }

  if (known) knownCount++; else unknownCount++;
  current++;

  if (current < queue.length) {
    showCard(current);
  } else {
    showSummary();
  }
}

$("btn-known").addEventListener("click", () => handleReview(true));
$("btn-unknown").addEventListener("click", () => handleReview(false));

function showSummary() {
  $("card-container").classList.add("hidden");
  $("controls").classList.add("hidden");
  $("count-known").textContent = knownCount;
  $("count-unknown").textContent = unknownCount;
  $("summary").classList.remove("hidden");
}

init();
```

**注意**：`GET /api/words?due=true` 回傳 `{ count, words: [...] }`，每個 word 有 `{ id, french, english, cefr_level }`。
